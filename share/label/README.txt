COMPILE:
	$ make

INSTALL:
	# make install

RUN:
	$ vision-artificial
			if it doesn't work:
	$ /usr/local/games/vision-artificial

UNINSTALL:
	# make uninstall


The program creates a folder '.vision-artificial/' at $HOME.  It
is used by the program to store saved images.  If you remove the program, you
should delete that folder manually.
